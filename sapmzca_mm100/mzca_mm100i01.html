MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.
    WHEN 'CANC'.
      PERFORM CLEAR_DATA.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'BTN_OK'.
      PERFORM BANPR_W_D USING '승인 처리 하시겠습니까?'
                               'A'
                               GO_ALV_GRID.
    WHEN 'BTN_NO'.
      PERFORM BANPR_W_D USING '반려 처리 하시겠습니까?'
                         'D'
                         GO_ALV_GRID.


  ENDCASE.

  PERFORM DATA_SELECT.
  GO_ALV_GRID->REFRESH_TABLE_DISPLAY( ).

ENDMODULE.
