MODULE EXIT INPUT.

  CASE OK_CODE.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.

ENDMODULE.

MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.
    WHEN 'BTN_SO' .
      PERFORM DISPLAY_DATA.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  DISPLAY_DATA  INPUT
*&---------------------------------------------------------------------*

MODULE DISPLAY_DATA INPUT.
  SELECT *
    FROM ZCA_MARD
    INNER JOIN ZCA_MARC
    ON ZCA_MARC~MATNR EQ ZCA_MARD~MATNR
    INTO CORRESPONDING FIELDS OF TABLE GT_DISPLAY.
ENDMODULE.
