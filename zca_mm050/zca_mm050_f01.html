FORM DISPLAY_DATA .
  CLEAR GT_DISPLAY.

  SELECT ZCA_EKPO~EBELN,
         ZCA_EKPO~EBELP,
         ZCA_EKPO~BANFN,
         ZCA_EKPO~MATNR,
         ZCA_EKPO~LGORT,
         ZCA_EKPO~NETPR,
         ZCA_EKPO~DMBTR,
         ZCA_EKPO~WAERS,
         ZCA_EKPO~MENGE,
         ZCA_EKPO~MEINH,
         ZCA_EKKO~MANDT,
         ZCA_EKKO~BEDAT,
         ZCA_EKKO~LTFNR,
         ZCA_EKKO~WERKS,
         ZCA_EKKO~BRTWR,
         ZCA_EKKO~ELIKZ,
         ZCA_EKKO~WEPOS,
         ZCA_EKKO~INDAT
    FROM ZCA_EKKO
    INNER JOIN ZCA_EKPO
    ON ZCA_EKKO~EBELN EQ ZCA_EKPO~EBELN
    WHERE ZCA_EKKO~LTFNR IN @SO_LTFNR
    AND   ZCA_EKKO~EBELN IN @SO_EBELN
    AND   BEDAT IN @SO_BEDAT
    AND   ZCA_EKKO~LVORM EQ ' '
    INTO CORRESPONDING FIELDS OF TABLE @GT_DISPLAY.

ENDFORM.

  IF ZCA_EKPO-EBELN IS INITIAL

  AND ZCA_EKPO-EBELP IS INITIAL.

      FROM ZCA_EKPO
      INTO CORRESPONDING FIELDS OF TABLE GT_DISPLAY.
  ENDIF.

ENDFORM.
