MODULE
  user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'TAB1'.

      SELECT *
INTO CORRESPONDING FIELDS OF TABLE gt_mseg
FROM zca_mkpf AS mkpf
INNER JOIN zca_mseg AS mseg
ON mkpf~mblnr = mseg~mblnr
AND mkpf~mjahr = mseg~mjahr
WHERE mseg~matnr IN so_matnr
AND mkpf~mjahr IN so_mjahr
AND mkpf~werks IN so_werks
AND mseg~bwart = '101'
        AND mseg~ernam IN so_ernam
        AND mseg~erdat IN so_erdat.

      my_tabstrip-activetab = ok_code.
    WHEN 'TAB2'.

      SELECT *
INTO CORRESPONDING FIELDS OF TABLE gt_mseg
FROM zca_mkpf AS mkpf
INNER JOIN zca_mseg AS mseg
ON mkpf~mblnr = mseg~mblnr
AND mkpf~mjahr = mseg~mjahr
WHERE mseg~matnr IN so_matnr
AND mkpf~mjahr IN so_mjahr
AND mkpf~werks IN so_werks
AND mseg~bwart = '261'
        AND mseg~ernam IN so_ernam
        AND mseg~erdat IN so_erdat.
      my_tabstrip-activetab = ok_code.

    WHEN 'TAB3'.

      SELECT *
INTO CORRESPONDING FIELDS OF TABLE gt_mseg
FROM zca_mkpf AS mkpf
INNER JOIN zca_mseg AS mseg
ON mkpf~mblnr = mseg~mblnr
AND mkpf~mjahr = mseg~mjahr
WHERE mseg~matnr IN so_matnr
AND mkpf~mjahr IN so_mjahr
AND mkpf~werks IN so_werks
AND mseg~bwart = '601'
        AND mseg~ernam IN so_ernam
        AND mseg~erdat IN so_erdat.

      my_tabstrip-activetab = ok_code.

    WHEN 'TAB4'.

      SELECT *
INTO CORRESPONDING FIELDS OF TABLE gt_mseg
FROM zca_mkpf AS mkpf
INNER JOIN zca_mseg AS mseg
ON mkpf~mblnr = mseg~mblnr
AND mkpf~mjahr = mseg~mjahr
WHERE mseg~matnr IN so_matnr
AND mkpf~mjahr IN so_mjahr
AND mkpf~werks IN so_werks
AND mseg~bwart = '301'
        AND mseg~ernam IN so_ernam
        AND mseg~erdat IN so_erdat.


      my_tabstrip-activetab = ok_code.


  ENDCASE.

ENDMODULE.
