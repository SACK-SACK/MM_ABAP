*& Module FILL_DYNNR_0100 OUTPUT
*&---------------------------------------------------------------------*
MODULE fill_dynnr_0100 OUTPUT.

  CASE my_tabstrip-activetab.
    WHEN 'TAB1'.
      " 첫 번째 탭의 Subscreen 번호는 0110
      dynnr = '0110'.

    WHEN 'TAB2'.
      " 두 번째 탭의 Subsscreen 번호는 0120
      dynnr = '0120'.

    WHEN 'TAB3'.
      " 두 번째 탭의 Subsscreen 번호는 0130
      dynnr = '0130'.

    WHEN 'TAB4'.
      " 두 번째 탭의 Subsscreen 번호는 0140
      dynnr = '0140'.

    WHEN OTHERS.
      "초기 기본값 설정
      my_tabstrip-activetab = 'TAB1'.
      dynnr = '0110'.
  ENDCASE.
ENDMODULE.
