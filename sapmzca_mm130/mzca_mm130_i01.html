MODULE USER_COMMAND_0100 INPUT.

*--------------------------------------------------------------------------------------
* 버튼 클릭시 작동
*--------------------------------------------------------------------------------------
  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'BTN_DIS'.
      PERFORM BTN_DIS.

    WHEN 'UPDATE'.
      PERFORM UPDATE.

    WHEN 'SAVE'.
      PERFORM SAVE.

    WHEN 'DELETE'.
      PERFORM DELETE.

    WHEN 'CREATE'.
      PERFORM CREATE_EINA_DATA.
      PERFORM CREATE_CELL.

  ENDCASE.

ENDMODULE.

    WHEN OTHERS.
  ENDCASE.
